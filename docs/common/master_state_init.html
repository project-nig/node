294    def extract_account_list_from_locking_script(self,op_elem,utxo):
295        account_list=[]
296        op_elem_account_list=[]
297        locking_script=utxo['locking_script']
298        locking_script_list = locking_script.split(" ")
299        #list of op_elem
300        # =>OP_SC is used to associate an account to a SmartContract
301        # =>OP_DEL_SC is used to deassociate an account from a SmartContract
302        #ex: OP_DUP OP_HASH160 31f2ac8088005412c7b031a6e342b17a65a48d01 OP_EQUAL_VERIFY OP_CHECKSIG OP_SC 47866ef83717f16c24cb246deeef1f75f798b8e5 
303        ##OP_SC 531c2e528bd177866f7213b192833846018686e5 OP_DEL_SC 3243223324GFDG
304        flag_for_adding=False
305        for element in locking_script_list:
306            #Step 1 list of account when there is no specific OP like OP_SC, OP_DEL_SC
307            if element.startswith("OP"):
308                pass
309            else:
310                account_list.append(element)
311            #Step 2 list for specific OP like OP_SC, OP_DEL_SC
312            if element.startswith(op_elem):flag_for_adding=True
313            elif flag_for_adding is True:
314                op_elem_account_list.append(element)
315                flag_for_adding=False
316        
317        if op_elem_account_list!=[]:return op_elem_account_list
318        else:
319            if op_elem=="OP_DEL_SC":return []
320            else:return account_list
