363    def run_smart_contract(self,expr):
364        self.result=None
365        if self.gas_ok_flag is True:
366            try:
367                #launch of the Smart Contract
368                start = datetime.now()
369                #print(f"===== expr:{expr}")
370                tree = ast.parse(expr)
371                eval_expr = ast.Expression(tree.body[-2].value)
372                eval_expr_memory = ast.Expression(tree.body[-1].value)
373                exec_expr = ast.Module(tree.body[:-2],type_ignores=[])
374                exec(compile(exec_expr, 'file', 'exec'))
375                self.result=eval(compile(eval_expr, 'file', 'eval'))
376                self.smart_contract_memory=eval(compile(eval_expr_memory, 'file', 'eval'))
377                end = datetime.now()
378                self.smart_contract_memory_size=getsizeof(tree)
379                self.smart_contract_memory_size+=getsizeof(self.code_source)
380                self.cpu=(end-start).total_seconds()
381                self.burn_gas()
382                logging.info(f"operation in {self.cpu} sec / memory {self.smart_contract_memory_size} / gas {int(self.gas)}")
383            except Exception as e:
384                #issue with the processing of the SmartContract
385                logging.info(f"**** ISSUE run_smart_contract: {self.smart_contract_account}")
386                logging.exception(e)
387                self.error_flag=True
388                self.error_code=e
389        else:
390            logging.info(f"**** WARNING not enough gas to run_smart_contract")
