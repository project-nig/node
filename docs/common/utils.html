  9def calculate_hash(data, hash_function: str = "sha256") -> str:
 10    """
 11    Calculate the hash of a string
 12    """
 13    data = bytearray(data, "utf-8")
 14    if hash_function == "sha256":
 15        h = SHA256.new()
 16        h.update(data)
 17        return h.hexdigest()
 18    if hash_function == "ripemd160":
 19        h = RIPEMD160.new()
 20        h.update(data)
 21        return h.hexdigest()

 24def normal_round(num, ndigits=0):
 25    """
 26    Rounds a float to the specified number of decimal places.
 27    num: the value to round
 28    ndigits: the number of digits to round to
 29    """
 30    if ndigits == 0:
 31        return int(num + 0.5)
 32    else:
 33        digit_value = 10 ** ndigits
 34        return int(num * digit_value + 0.5) / digit_value

 55def dict_replace_value(d: dict) -> dict:
 56    """
 57    Function used by clean_request function
 58    """
 59    x = {}
 60    for k, v in d.items():
 61        if v=="true":v=True
 62        elif v=="false":v=False
 63        elif v=="none":v=None
 64        elif isinstance(v, dict):
 65            v = dict_replace_value(v)
 66        elif isinstance(v, list):
 67            v = list_replace_value(v)
 68        x[k] = v
 69    return x
